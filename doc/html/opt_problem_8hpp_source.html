175 template <typename _Scalar> int
176 OptProblem<_Scalar>::addLinConstraint( BOUND bound_type, Scalar lower_bound, Scalar upper_bound, std::vector<Scalar> coeffs, LINEARITY c_lin )
177 {
178  // usage:
179  // coeffs[0] * x_0 + coeffs[1] * x_1 ... + coeffs[n] * x_n >= lower_bound , bound_type == BOUND::GREATER_EQ
180  // OR coeffs[0] * x_0 + coeffs[1] * x_1 ... + coeffs[n] * x_n <= upper_bound , bound_type == BOUND::LESS_EQ
181  // OR coeffs[0] * x_0 + coeffs[1] * x_1 ... + coeffs[n] * x_n = lower_bound = upper_bound, bound_type == BOUND::FIXED
182  if ( coeffs.size() != getVarCount() )
183  {
184  std::cerr << "[" << __func__ << "]: " << "A line in the constraints matrix A has to be varCount " << getVarCount() << " long, not " << coeffs.size() << std::endl;
185  return EXIT_FAILURE;
186  }
187 
188  // prepare
189  SparseMatrix row_vector( 1, coeffs.size() );
190  // add coeffs from new line
191  for ( size_t col = 0; col != coeffs.size(); ++col )
192  {
193  // add non-zero elements to sparse representation
194  if ( coeffs[col] != Scalar(0) )
195  {
196  row_vector.insert( 0, col ) = coeffs[col];
197  }
198  }
199 
200  // work
201  return this->addConstraint( bound_type, lower_bound, upper_bound, &row_vector, c_lin );
202 }
