305  template <typename _Scalar> int
306  OptProblem<_Scalar>::_parseAuxFile( std::string const& aux_file_path )
307  {
308  int err = EXIT_SUCCESS;
309 
310  // open file
311  std::ifstream aux_file( aux_file_path );
312  if ( !aux_file.is_open() )
313  {
314  std::cerr << "[" << __func__ << "]: " << "Coult not open aux_file " << aux_file_path << std::endl;
315  return EXIT_FAILURE;
316  }
317 
318  // parse lines
319  int vars = -1, // "uninited"
320  constraints = -1; // "uninited"
321  int lid = -1; // -1 == there's an extra line to read first, the header
322  std::string line; // tmp storage of read line
323  while ( getline(aux_file, line) )
324  {
325  // skip comment
326  if ( line[0] == '#') continue;
327 
328  // parse line
329  std::istringstream iss( line );
330  std::string word;
331  int word_id = 0; // line parse state
332  Variable tmp; // output storage variable
333  while ( std::getline(iss, word, ',') )
334  {
335  if ( vars < 0 ) // header line
336  {
337  vars = atoi( word.c_str() ); // parse number of variables
338  std::getline(iss, word, ',');
339  constraints = atoi( word.c_str() ); // parse number of constraints
340  std::getline(iss, word, ',');
341  this->_cfix = atof( word.c_str() ); // parse constant objective function bias
342 
343  std::cout << "bias is now " << this->getObjectiveBias()
344  << " reading " << vars << " vars and " << constraints << " constraints"
345  << std::endl;
346 
347  std::getline( iss, word ); // clear rest of line
348  }
349  else // variable or constraint line
350  {
351  if ( lid < vars ) // variable line
352  {
353  switch ( word_id ) // bound_type, var_type, lower_boud, upper_bound, linearity
354  {
355  case 0: // Variable bound type
356  tmp._bkx = static_cast<BOUND>( atoi(word.c_str()) );
357  break;
358  case 1: // Variable type (cont/integer/binary)
359  tmp._type_x = static_cast<VAR_TYPE>( atoi(word.c_str()) );
360  break;
361  case 2: // Variable lower bound
362  tmp._blx = atof( word.c_str() );
363  break;
364  case 3: // Variable upper bound
365  tmp._bux = atof( word.c_str() );
366  break;
367  case 4: // Variable linearity
368  tmp._lin_x = static_cast<LINEARITY>( atoi(word.c_str()) );
369  break;
370  default:
371  std::cerr << "[" << __func__ << "]: " << "Unknown VAR word_id..." << word_id << std::endl;
372  break;
373  } //...switch word_id
374  } //...if variable
375  else // constraint line
376  {
377  switch( word_id ) // bound_type, lower_boud, upper_bound, linearity
378  {
379  case 0: // Constraint bound type
380  tmp._bkx = static_cast<BOUND>( atoi(word.c_str()) );
381  break;
382  case 1: // Constraint lower bound
383  tmp._blx = atof( word.c_str() );
384  break;
385  case 2: // Constraint upper bound
386  tmp._bux = atof( word.c_str() );
387  break;
388  case 3: // Constraint linearity
389  tmp._lin_x = static_cast<LINEARITY>( atoi(word.c_str()) );
390  break;
391  default:
392  std::cerr << "[" << __func__ << "]: " << "Unknown CNSTR word_id..." << word_id << ", word: " << word << std::endl;
393  break;
394  } //...switch word_id
395  } //...if constraint
396 
397  // increment line parse status
398  ++word_id;
399  } //... non-header line
400  } //...while line has more words
401 
402  // add variable/constraint
403  if ( lid >= 0 )
404  {
405  if ( lid < vars ) this->addVariable ( tmp ); // variable
406  else this->addConstraint( tmp ); // constraint
407  } // if non-header-line
408 
409  // increase read line count (did not count comments)
410  ++lid;
411  } //... while file has more lines
412 
413  // close file
414  aux_file.close();
415 
416  return err;
417  } //...OptProblem::_parseAuxFile()
